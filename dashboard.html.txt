        function filterAndRender() {
            const search = document.getElementById('search-input').value.toLowerCase();
            const roleFilter = document.getElementById('role-filter').value;
            const perfFilter = document.getElementById('performance-filter').value;

            let filtered = employees;

            // Пошук
            if (search) {
                filtered = filtered.filter(emp => 
                    emp.name.toLowerCase().includes(search) || 
                    (emp.improvementFocus && emp.improvementFocus.toLowerCase().includes(search))
                );
            }

            // Фільтр по ролі
            if (roleFilter) {
                filtered = filtered.filter(emp => emp.role === roleFilter);
            }

            // Фільтр по рівню performance
            if (perfFilter === 'high') {
                filtered = filtered.filter(emp => emp.performanceScore >= 85);
            } else if (perfFilter === 'medium') {
                filtered = filtered.filter(emp => emp.performanceScore >= 70 && emp.performanceScore < 85);
            } else if (perfFilter === 'low') {
                filtered = filtered.filter(emp => emp.performanceScore < 70);
            }

            // Сортування
            filtered.sort((a, b) => {
                let valA = a[currentSort.key];
                let valB = b[currentSort.key];
                
                if (typeof valA === 'string') {
                    return currentSort.direction === 'asc' 
                        ? valA.localeCompare(valB) 
                        : valB.localeCompare(valA);
                } else {
                    return currentSort.direction === 'asc' ? valA - valB : valB - valA;
                }
            });

            renderTable(filtered);
        }

        function sortTable(key) {
            if (currentSort.key === key) {
                currentSort.direction = currentSort.direction === 'asc' ? 'desc' : 'asc';
            } else {
                currentSort.key = key;
                currentSort.direction = key === 'name' ? 'asc' : 'desc';
            }
            filterAndRender();
        }

            const newData = {
                name: name,
                role: document.getElementById('modal-role').value,
                salesTargetPercent: parseInt(document.getElementById('modal-sales').value) || 0,
                customerSatisfaction: parseFloat(document.getElementById('modal-satisfaction').value) || 0,
                attendance: parseInt(document.getElementById('modal-attendance').value) || 0,
                performanceScore: parseInt(document.getElementById('modal-performance').value) || 0,
                itemsPerHour: parseInt(document.getElementById('modal-items').value) || 0,
                improvementFocus: document.getElementById('modal-focus').value.trim()
            };

            if (currentEmployeeId) {
                // Редагування
                const index = employees.findIndex(e => e.id === currentEmployeeId);
                employees[index] = { ...employees[index], ...newData };
            } else {
                // Новий співробітник
                const newId = Math.max(0, ...employees.map(e => e.id)) + 1;
                employees.push({
                    id: newId,
                    department: 'Торговий зал',
                    lastReview: new Date().toISOString().split('T')[0],
                    ...newData
                });
            }

            saveEmployees();
            closeModal();
            filterAndRender();
            renderKPIs();
        }

        function markForCoaching(id) {
            const emp = employees.find(e => e.id === id);
            if (!emp) return;

            if (confirm(`Відмітити ${emp.name} для коучингу?`)) {
                emp.improvementFocus = emp.improvementFocus 
                    ? emp.improvementFocus + ' | Потребує коучингу' 
                    : 'Потребує коучингу (низькі показники)';
                
                saveEmployees();
                filterAndRender();
                
                // Показуємо рекомендацію
                setTimeout(() => {
                    alert(`Рекомендація для ${emp.name}:\n• Провести індивідуальну розмову\n• Поставити чіткі цілі на тиждень\n• Контролювати прогрес через 7 днів`);
                }, 300);
            }
        }

        // ==================== ІНІЦІАЛІЗАЦІЯ ====================
        function init() {
            loadEmployees();
            
            // Початковий рендер
            filterAndRender();
            renderKPIs();
            
            // Додаємо слухачі для сортування (вже є onclick в HTML)
            // Додатковий обробник для Enter в пошуку
            document.getElementById('search-input').addEventListener('keypress', function(e) {
                if (e.key === 'Enter') filterAndRender();
            });
            
            // Початкове сортування за performance
            currentSort = { key: 'performanceScore', direction: 'desc' };
        }

        // Запускаємо додаток
        window.onload = init;
    </script>
